train_im <- train
tree_adj_fun <- function(im) {
  train_im <- train[, im]
  WAUS.tree <- tree(RainTomorrow ~ ., data = train_im)
  
  tree.pred <- predict(WAUS.tree, test.x, type='class')
  tree.cm <- table(observed=test.y, predicted=tree.pred)
  tree.ac <- (tree.cm[1, 1] + tree.cm[2, 2])/sum(tree.cm)
  
  tree.pred.conf <- predict(WAUS.tree, test.x, type='vector')
  treepred <- ROCR::prediction(tree.pred.conf[, 2], test.y)
  treeperf <- ROCR::performance(treepred, 'tpr', 'fpr')
  tree.auc <- ROCR::performance(treepred, 'auc')@y.values[[1]]
  
  ret1 <- c(tree.ac, tree.auc)
  
  # prune tree
  testtree <- cv.tree(WAUS.tree, FUN=prune.misclass)
  size <- testtree$size[which.min(testtree$dev)]
  
  prune.tree <- prune.misclass(WAUS.tree, best=size)
  prune.tree.pred <- predict(prune.tree, test.x, type='class')

  prune.tree.cm <- table(observed=test.y, predicted=prune.tree.pred)
  prune.tree.ac <- (prune.tree.cm[1, 1] + prune.tree.cm[2, 2])/sum(prune.tree.cm)
  prune.tree.ac

  prune.tree.pred.conf <- predict(prune.tree, test.x, type='vector')
  prune.treepred <- ROCR::prediction(prune.tree.pred.conf[, 2], test.y)
  prune.treeperf <- ROCR::performance(prune.treepred, 'tpr', 'fpr')
  prune.tree.auc <- ROCR::performance(prune.treepred, 'auc')@y.values[[1]]
  prune.tree.auc
  
  ret2 <- c(prune.tree.ac, prune.tree.auc)
  
  ret <- rbind(ret1, ret2)
  colnames(ret) <- c('Accuracy', 'AUC')
  rownames(ret) <- c('Raw', 'Pruned')
  return(ret)

  train_im <- train[, im]
  WAUS.tree <- tree(RainTomorrow ~ ., data = train_im)
  
  tree.pred <- predict(WAUS.tree, test.x, type='class')
  tree.cm <- table(observed=test.y, predicted=tree.pred)
  tree.ac <- (tree.cm[1, 1] + tree.cm[2, 2])/sum(tree.cm)
  
  tree.pred.conf <- predict(WAUS.tree, test.x, type='vector')
  treepred <- ROCR::prediction(tree.pred.conf[, 2], test.y)
  treeperf <- ROCR::performance(treepred, 'tpr', 'fpr')
  tree.auc <- ROCR::performance(treepred, 'auc')@y.values[[1]]
  
  ret1 <- c(tree.ac, tree.auc)
  
  # prune tree
  testtree <- cv.tree(WAUS.tree, FUN=prune.misclass)
  size <- testtree$size[which.min(testtree$dev)]
  
  prune.tree <- prune.misclass(WAUS.tree, best=size)
  prune.tree.pred <- predict(prune.tree, test.x, type='class')

  prune.tree.cm <- table(observed=test.y, predicted=prune.tree.pred)
  prune.tree.ac <- (prune.tree.cm[1, 1] + prune.tree.cm[2, 2])/sum(prune.tree.cm)
  prune.tree.ac

  prune.tree.pred.conf <- predict(prune.tree, test.x, type='vector')
  prune.treepred <- ROCR::prediction(prune.tree.pred.conf[, 2], test.y)
  prune.treeperf <- ROCR::performance(prune.treepred, 'tpr', 'fpr')
  prune.tree.auc <- ROCR::performance(prune.treepred, 'auc')@y.values[[1]]
  prune.tree.auc
  
  ret2 <- c(prune.tree.ac, prune.tree.auc)
  
  ret <- rbind(ret1, ret2)
  colnames(ret) <- c('Accuracy', 'AUC')
  rownames(ret) <- c('Raw', 'Pruned')
  
  return(ret)
}
